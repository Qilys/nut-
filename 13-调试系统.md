```squirrel
function drawCustomUI_debug(obj)
```

## 状态信息调试

### 状态信息处理函数

```squirrel
function stateinfo(obj)
{
    local attackIndex = obj.getAttackIndex();        // 得到攻击ID
    local sksubstate = obj.getSkillSubState();       // 得到技能子状态
    local skillindex = obj.getCurrentSkillIndex();   // 得到技能ID

    // 存储状态信息到vector
    obj.getVar("stateinfo1").clear_vector();
    obj.getVar("stateinfo1").push_vector(skillindex);    // 存入技能ID
    obj.getVar("stateinfo1").push_vector(sksubstate);    // 存入子状态ID

    obj.getVar("stateinfo2").clear_vector();
    obj.getVar("stateinfo2").push_vector(attackIndex);   // 存入攻击ID

    // 获取动画ID（需要循环匹配）
    local ani = obj.sq_GetCurrentAni();              // 获取当前的ani
    for(local i=0; i<300; i++)                       // 循环300次，匹配i=当前ani
    {
        if(sq_IsSameAni(ani, sq_GetCustomAni(obj, i)))  // 判断是否相同
        {
            obj.getVar("stateinfo2").push_vector(i);     // 如果是相同的，就把i值放入向量中
            break;                                       // 跳出循环
        }
    }

    // 绘制信息
    sqr_Print2(obj, "AtkID=", 20, 360);             // 绘制AtkID字符
    sqr_Print2(obj, "AniID=", 20, 378);             // 绘制AniID字符
    sqr_Print(obj, "stateinfo2", 85, 360);          // 绘制stateinfo2向量的值

    sqr_Print2(obj, "SkillID=", 20, 250);
    sqr_Print2(obj, "SkillSub=", 20, 268);
    sqr_Print(obj, "stateinfo1", 105, 250);
}
```

### 扩展状态信息调试

```squirrel
function stateinfo(obj)

}
```

## 绘制函数

### sqr_Print2 - 绘制字符串和数字

```squirrel
function sqr_Print2(obj, Value, x, y)
{
    // 检查类型，必须是字符串或整数
    if (type(Value) != "string" && type(Value) != "integer") {
        return;
    }

    local len = Value.len();                         // 字符串取长度
    local anipath = "sqr/debug/animation/ui_font.ani";  // 设置ani路径
    
    // 字符映射表（26个字母+数字+特殊字符）
    local into = {
        ["a"]=0,["b"]=1,["c"]=2,["d"]=3,["e"]=4,["f"]=5,["g"]=6,["h"]=7,["i"]=8,["j"]=9,
        ["k"]=10,["l"]=11,["m"]=12,["n"]=13,["o"]=14,["p"]=15,["q"]=16,["r"]=17,["s"]=18,
        ["t"]=19,["u"]=20,["v"]=21,["w"]=22,["x"]=23,["y"]=24,["z"]=25,
        ["A"]=26,["B"]=27,["C"]=28,["D"]=29,["E"]=30,["F"]=31,["G"]=32,["H"]=33,["I"]=34,
        ["J"]=35,["K"]=36,["L"]=37,["M"]=38,["N"]=39,["O"]=40,["P"]=41,["Q"]=42,["R"]=43,
        ["S"]=44,["T"]=45,["U"]=46,["V"]=47,["W"]=48,["X"]=49,["Y"]=50,["Z"]=51,
        ["0"]=52,["1"]=53,["2"]=54,["3"]=55,["4"]=56,["5"]=57,["6"]=58,["7"]=59,["8"]=60,["9"]=61,
        ["="]=68
    };

    // 根据字符串长度循环
    for(local i=0; i<len; i+=1)
    {
        local val = into[Value.slice(i, i+1)];           // 取出字符对应的帧数
        local ani = obj.getVar().GetAnimationMap("ui_0", anipath);  // 自定义ani
        local valInt = val.tointeger();                  // 转换为整数
        sq_SetAnimationFrameIndex(ani, valInt);          // 设置ani帧
        sq_drawCurrentFrame(ani, x+i*10, y, false);      // 绘制当前帧
    }
}
```

### sqr_Print - 绘制向量中的数值

```squirrel
function sqr_Print(obj, vartext, x, y)
{
    local count = obj.getVar(vartext).size_vector();     // 获取向量元素数量
    
    for(local a=0; a<count; a++)                         // 根据元素数量循环
    {
        local Value = obj.getVar(vartext).get_vector(a).tostring();  // 获取元素并转为字符串
        local len = Value.len();                         // 得到长度
        local anipath = "sqr/debug/animation/ui_font.ani";
        
        // 字符映射表（同上）
        local into = { /* ... */ };
        
        for(local i=0; i<len; i++)
        {
            local val = into[Value.slice(i, i+1)];       // 按顺序依次取出字符
            local ani = obj.getVar().GetAnimationMap("ui_0", anipath);
            local valInt = val.tointeger();
            sq_SetAnimationFrameIndex(ani, valInt);
            sq_drawCurrentFrame(ani, x+i*10, y+a*18, false);  // y坐标增加偏移
        }
    }
}
```

### addNumberUi - 数字UI添加函数

```squirrel
function addNumberUi(obj, vartext, xpos, xadd, ypos, yadd)
{
    local count = obj.getVar(vartext).size_vector();
    
    for(local a=0; a<count; a++)
    {
        local Value = obj.getVar(vartext).get_vector(a).tostring();
        local len = Value.len();                         // 得到长度

        // 分割实际数字
        for(local i=0; i<len; i++)
        {
            local son = Value.slice(i, i+1);             // 按顺序依次取出字符
            local uipath = getNumberAnimation(obj, son); // 得到对应数字ani
            sq_AnimationProc(uipath);
            sq_drawCurrentFrame(uipath, xpos+i*xadd, ypos+a*yadd, false);  // 绘制ui
        }
    }
}
```

### getNumberAnimation - 获取数字动画

```squirrel
function getNumberAnimation(obj, son)
{
    local uipath = null;
    switch(son)
    {
        case "-":
        case ".": uipath = obj.getVar().GetAnimationMap("ui_", "sqr/debug/animation/ui_.ani"); break;
        case "0": uipath = obj.getVar().GetAnimationMap("ui_0", "sqr/debug/animation/ui_0.ani"); break;
```

## 调试技巧

### 获取动画ID

由于动画ID不能直接获取，需要通过循环匹配来获取：

```squirrel
local ani = obj.sq_GetCurrentAni();              // 获取当前动画
for(local i=0; i<300; i++)                       // 循环匹配
{
    if(sq_IsSameAni(ani, sq_GetCustomAni(obj, i)))
    {
        // i 就是当前的动画ID
        print("Current AniID: " + i);
        break;
    }
}
```

```squirrel
// 存储数据到vector
```
