# UI 绘制

> 本文档说明UI绘制相关的函数和方法

## UI绘制函数

### drawMainCustomUI_* - 画城镇和副本UI

城镇中和副本中都可以画，图层较低。

```squirrel
function drawMainCustomUI_职业名(obj)
{
    if (!obj) return;
    local sq_var = obj.getVar();
    
    // 获取动画
    local hud_ani = sq_var.GetAnimationMap("hud_name", "路径/ui/hud.ani");
    
    // 绘制
    sq_DrawSpecificFrame(
        hud_ani,    // 动画
        x,          // x坐标
        y,          // y坐标
        false,      // 是否翻转
        0,          // 帧数
        false,      // 
        1.0         // 缩放
    );
}
```

### drawCustomUI_* - 画副本中的UI

只能在副本中进行画UI。

```squirrel
function drawCustomUI_职业名(obj)
{
    if (!obj) return;
    if (!obj.isInBattle()) return;  // 非战斗状态直接返回
    
    // 鼠标相关
    local isLock = sq_IsClipCursor();
    if (!isLock)
    {
        if (!sq_IsVisibleCursor())
            sq_SetVisibleCursor(true);
        return;
    }
    if (sq_IsVisibleCursor())
        sq_SetVisibleCursor(false);
    
    // 获取动画
    local ani = obj.getVar().GetAnimationMap("NORMAL", "路径/ui/cursor.ani");
    
    // 根据条件改变动画
    if (condition)
        ani = obj.getVar().GetAnimationMap("OTHER", "路径/ui/other.ani");
    
    // 绘制
    if (ani)
    {
        local posX = IMouse.GetXPos();
        local posY = IMouse.GetYPos();
        sq_AnimationProc(ani);
        sq_drawCurrentFrame(ani, posX, posY, false);
    }
}
```

### drawAppend_* - 画人物当前图层的UI

初始坐标为人物的偏移。

```squirrel
// isOver参数为true画身前 false画身后
function drawAppend_职业名(obj, isOver, x, y)
{
    if (!obj.isMyControlObject()) return 0;
    
    local sq_var = obj.getVar();
    local ani = sq_var.GetAnimationMap("name", "路径/ani.ani");
    
    if (ani && isOver)  // 身前
    {
        sq_AnimationProc(ani);
        sq_drawCurrentFrame(ani, x, y, false);
    }
    
    return 0;
}
```

### prepareDraw_* - 准备画

```squirrel
function prepareDraw_职业名(obj)
{
    if (!obj) return;
    
    if (condition)
        return S_FLOW_NORMAL;
    
    return S_FLOW_PRIEST;
}
```

## 绘制函数

### sq_DrawSpecificFrame - 绘制指定帧

```squirrel
sq_DrawSpecificFrame(
    animation,    // 动画对象
    x,            // x坐标
    y,            // y坐标
    isFlip,       // 是否翻转
    frameIndex,   // 帧数
    false,        // 
    scale         // 缩放（1.0为原始大小）
);
```

### sq_DrawSpecificFrameEffect_SIMPLE - 绘制带效果的帧

```squirrel
sq_DrawSpecificFrameEffect_SIMPLE(
    animation,    // 动画对象
    x,            // x坐标
    y,            // y坐标
    frameIndex,   // 帧数
    rgb,          // RGB颜色
    alpha,        // 透明度
    true          // 
);

// RGB和Alpha颜色
local rgb = sq_RGB(255, 0, 0);        // RGB颜色
local alpha = sq_ALPHA(255);          // 透明度
```

### sq_drawCurrentFrame - 绘制当前帧

```squirrel
sq_AnimationProc(ani);              // 更新动画
sq_drawCurrentFrame(ani, x, y, isFlip);  // 绘制当前帧
```

## UI示例

### 进度条绘制示例

```squirrel
function drawMainCustomUI_职业名(obj)
{
    if (!obj) return;
    local sq_var = obj.getVar();
    
    local hud_gauge = sq_var.GetAnimationMap("gauge", "路径/ui/gauge.ani");
    local x = 626;
    local y = 588;
    
    // 计算比率
    local currentValue = getCurrentValue(obj);
    local maxValue = getMaxValue(obj);
    local rate = currentValue.tofloat() / maxValue.tofloat();
    
    // 设置比率
    hud_gauge.setImageRate(rate, 1.0);
    
    // 绘制
    local rgb = sq_RGB(255, 255, 255);
    local alpha = sq_ALPHA(255);
    sq_DrawSpecificFrameEffect_SIMPLE(hud_gauge, x, y, 0, rgb, alpha, true);
}
```

### 跟随鼠标的UI示例

```squirrel
function drawCustomUI_职业名(obj)
{
    if (!obj) return;
    if (!obj.isInBattle()) return;
    
    // 根据技能状态改变动画
    local ani = null;
    local usingSkillIndex = getSkillIndex(obj);
    
    switch (usingSkillIndex)
    {
        case SKILL_TYPE_1:
            ani = obj.getVar().GetAnimationMap("TYPE1", "路径/cursor1.ani");
            break;
        case SKILL_TYPE_2:
            ani = obj.getVar().GetAnimationMap("TYPE2", "路径/cursor2.ani");
            break;
    }
    
    // 绘制在鼠标位置
    if (ani)
    {
        local posX = IMouse.GetXPos();
        local posY = IMouse.GetYPos();
        sq_AnimationProc(ani);
        sq_drawCurrentFrame(ani, posX, posY, false);
    }
}
```

### 角色图层UI示例

```squirrel
function drawAppend_职业名(obj, isOver, x, y)
{
    if (!obj.isMyControlObject()) return 0;
    
    local sq_var = obj.getVar();
    
    // 身前UI
    if (isOver)
    {
        local frontAni = sq_var.GetAnimationMap("front", "路径/front.ani");
        if (frontAni)
        {
            sq_AnimationProc(frontAni);
            sq_drawCurrentFrame(frontAni, x, y, false);
        }
    }
    // 身后UI
    else
    {
        local backAni = sq_var.GetAnimationMap("back", "路径/back.ani");
        if (backAni)
        {
            sq_AnimationProc(backAni);
            sq_drawCurrentFrame(backAni, x, y, false);
        }
    }
    
    return 0;
}
```

## 颜色和效果

### RGB颜色

```squirrel
local rgb = sq_RGB(255, 0, 0);  // 红色 (R, G, B)
local rgb = sq_RGB(0, 255, 0);  // 绿色
local rgb = sq_RGB(0, 0, 255);  // 蓝色
local rgb = sq_RGB(255, 255, 255);  // 白色
```

### Alpha透明度

```squirrel
local alpha = sq_ALPHA(255);  // 完全不透明 (0-255)
local alpha = sq_ALPHA(128);  // 半透明
local alpha = sq_ALPHA(0);    // 完全透明
```

### RGBA颜色

```squirrel
local rgba = sq_RGBA(255, 0, 0, 255);  // (R, G, B, A)
```
