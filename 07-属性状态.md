# 属性状态

> 本文档说明属性状态系统的使用方法

## 属性类型常量

详见 [12-常量定义.md](./12-常量定义.md) 中的 `CHANGE_STATUS_TYPE_*` 常量定义。

---

## 创建属性状态

### 创建基础属性状态

```squirrel
// 参数1：属性类型
// 参数2：false为增加数值，true为增加百分比数值
// 参数3：增加的数值
// 参数4：时间（0为无限）
local ap = sq_CreateChangeStatus(
    CHANGE_STATUS_TYPE_ELEMENT_TOLERANCE_FIRE, 
    false, 
    1000.0, 
    0
);

if (ap)
{
    ap.getAppendageInfo().setValidTime(10000);  // 设置有效时间
    ap.sq_Append(obj, obj);                     // 设置属性的父对象
}
```

---

## 附加可以看到图标的属性

这种方式创建的属性会在UI上显示图标。

```squirrel
// 先附加一个ap
local appendage = CNSquirrelAppendage.sq_AppendAppendage(
    obj, obj, SKILL_ID, false, 
    "路径/ap_xxx.nut", false
);

// 得到或创建属性状态
local change_appendage = appendage.sq_getChangeStatus("changeStatus");
if(!change_appendage)
    change_appendage = appendage.sq_AddChangeStatus(
        "changeStatus",           // 名称
        obj,                      // 对象1
        obj,                      // 对象2
        500000,                   // 有效时间
        CHANGE_STATUS_TYPE_PHYSICAL_CRITICAL_HIT_RATE,  // 属性类型
        false,                    // false为增加数值，true为增加百分比
        1000                      // 数值
    );
```

## 附加看不到图标的属性

这种方式创建的属性不会显示额外图标（被动技能已有图标的情况下使用）。

```squirrel
local change_appendage = appendage.sq_getChangeStatus("ele_atk_water");
if(!change_appendage)
    change_appendage = appendage.sq_AddChangeStatusAppendageID(
        obj, obj, 
        0,                        // 
        CHANGE_STATUS_TYPE_ELEMENT_ATTACK_WATER,  // 属性类型
        false,                    // false为增加数值，true为增加百分比
        registValue,              // 数值
        APID_COMMON               // AppendageID
    );

if(change_appendage)
{
    change_appendage.clearParameter();
    change_appendage.addParameter(
        CHANGE_STATUS_TYPE_ELEMENT_ATTACK_WATER, 
        false, 
        registValue.tofloat()
    );
}
```

## 属性参数操作

### 清除和添加参数

```squirrel
change_appendage.clearParameter();              // 清除参数
change_appendage.addParameter(
    CHANGE_STATUS_TYPE_XXX,  // 属性类型
    false,                   // false为数值，true为百分比
    value.tofloat()          // 数值
);
```

## 附加属性设置ID

### 使用ID附加（不可重复）

```squirrel
// 方式1：基础附加
ap.sq_Append(obj, obj);

// 方式2：带ID附加（不可重复附加）
ap.sq_Append(obj, obj, APID_SKILL_LIGHT_ENCHANT_WEAPON);

// 方式3：带ID和叠加次数
ap.sq_Append(object, obj, APID_AT_MAGE_ELEMENT_SHIELD, 1, null);
// 参数4：可叠加次数
// 参数5：不明
```

### 使用CNSquirrelAppendage附加

```squirrel
// 方式1：是否buff
CNSquirrelAppendage.sq_Append(ap, obj, obj, true);
// 参数4：true为buff（不能带过图），false可以带过图

// 方式2：带ID
CNSquirrelAppendage.sq_AppendAppendageID(
    ap, obj, obj, 
    APID_SKILL_LIGHT_ENCHANT_WEAPON, 
    false
);
```

## 删除属性

### 按ID删除属性

```squirrel
sq_RemoveChangeStatus(obj, APID_AT_MAGE_ELEMENT_SHIELD);
```

## 转换率

当需要设置HP/MP等属性时，可能需要使用转换率来获得正确的数值。

```squirrel
// 先得到转换率
local convert_rate = sq_GetAbilityConvertRate(obj, CONVERT_TABLE_TYPE_HP);

// 得到正确的数值
local correct_value = original_value.tofloat() * convert_rate.tofloat();
```

### 转换常量

```squirrel
CONVERT_TABLE_TYPE_DAMAGE <- 0              // 伤害
CONVERT_TABLE_TYPE_DEFENSE <- 1             // 防御
CONVERT_TABLE_TYPE_HP <- 2                  // HP MAX
CONVERT_TABLE_TYPE_PHYSICAL_ATTACK <- 3     // 力量
CONVERT_TABLE_TYPE_PHYSICAL_DEFENSE <- 4    // 体力
CONVERT_TABLE_TYPE_MAGICAL_ATTACK <- 5      // 智力
CONVERT_TABLE_TYPE_MAGICAL_DEFENSE <- 6     // 精神
CONVERT_TABLE_TYPE_PHYSICAL_ABSOLUTE_DAMAGE <- 7   // 物理绝对伤害
CONVERT_TABLE_TYPE_PHYSICAL_DAMAGE_REDUCE <- 8     // 物理伤害减少
CONVERT_TABLE_TYPE_MAGICAL_ABSOLUTE_DAMAGE <- 9    // 魔法绝对伤害
CONVERT_TABLE_TYPE_MAGICAL_DAMAGE_REDUCE <- 10     // 魔法伤害减少
CONVERT_TABLE_TYPE_ACTIVESTATUS_DAMAGE_REDUCE <- 11 // 异常状态伤害减少
CONVERT_TABLE_TYPE_MP <- 12                 // MP MAX
CONVERT_TABLE_TYPE_SKILL_POWER <- 13        // 技能伤害
```

## 完整示例

### 创建带图标的属性状态

```squirrel
// 1. 附加AP
local appendage = CNSquirrelAppendage.sq_AppendAppendage(
    obj, obj, SKILL_ID, false, 
    "路径/ap_xxx.nut", false
);

// 2. 创建属性状态
local change_appendage = appendage.sq_getChangeStatus("myStatus");
if(!change_appendage)
{
    local skillLevel = sq_GetSkillLevel(obj, SKILL_ID);
    local value = sq_GetLevelData(obj, SKILL_ID, 0, skillLevel);
    
    change_appendage = appendage.sq_AddChangeStatus(
        "myStatus",
        obj, obj,
        30000,                              // 30秒
        CHANGE_STATUS_TYPE_ATTACK_SPEED,   // 攻击速度
        false,                              // 数值型
        value                               // 数值
    );
}

// 3. 更新参数
if(change_appendage)
{
    change_appendage.clearParameter();
    change_appendage.addParameter(
        CHANGE_STATUS_TYPE_ATTACK_SPEED,
        false,
        newValue.tofloat()
    );
}
```

### 创建无图标的属性状态

```squirrel
local change_appendage = appendage.sq_getChangeStatus("hiddenStatus");
if(!change_appendage)
{
    change_appendage = appendage.sq_AddChangeStatusAppendageID(
        obj, obj, 0,
        CHANGE_STATUS_TYPE_MOVE_SPEED,
        false,
        speedValue,
        APID_COMMON
    );
}

if(change_appendage)
{
    change_appendage.clearParameter();
    change_appendage.addParameter(
        CHANGE_STATUS_TYPE_MOVE_SPEED,
        false,
        speedValue.tofloat()
    );
}
```
