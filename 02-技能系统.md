# 技能系统

> 本文档说明技能相关的函数和事件

## 技能使用相关

### useSkill_before_* - 使用技能前

在技能使用前执行，可以修改技能消耗等。

```squirrel
function useSkill_before_职业名(obj, skillIndex, consumeMp, consumeItem, oldSkillMpRate)
{
    if(!obj) return false;
    
    // 修改MP消耗率
    local appendage = obj.GetSquirrelAppendage("路径/ap_xxx.nut");
    if(appendage && appendage.isValid())
    {
        local skillLevel = obj.sq_GetSkillLevel(SKILL_ID);
        local mpComsumeRate = sq_GetLevelData(obj, SKILL_ID, 0, skillLevel);
        local newMpRate = oldSkillMpRate.tofloat() * (100 + mpComsumeRate.tofloat()) / 100;
        obj.setSkillMpRate(skillIndex, newMpRate.tofloat());
    }
    
    return true;
}
```

### useSkill_after_* - 使用技能后

```squirrel
function useSkill_after_职业名(obj, skillIndex, consumeMp, consumeItem, oldSkillMpRate)
{
    if(!obj) return false;
    // 恢复原始MP消耗率
    obj.setSkillMpRate(skillIndex, oldSkillMpRate.tofloat());
    return true;
}
```

### onChangeSkillEffect_* - 当前技能效果

接收技能效果包，处理技能效果数据。

```squirrel
function onChangeSkillEffect_职业名(obj, skillIndex, reciveData)
{
    if(!obj) return;
    
    if(skillIndex == SKILL_ID)
    {
        local skill = sq_GetSkill(obj, SKILL_ID);
        skill.resetCurrentCoolTime();
        skill.setSealActiveFunction(true);
        // ... 其他处理
    }
}
```

## 技能控制

### setEnableCancelSkill_* - 普攻强制施放技能

```squirrel
function setEnableCancelSkill_职业名(obj, isEnable)
{
    if(!obj) return false;
    if(!obj.isMyControlObject()) return false;
    if(!isEnable) return true;
    
    obj.setSkillCommandEnable(SKILL_ID, isEnable);
    return true;
}
```

### flushCommandEnable_* - 刷新技能按键状态

这里可以设置技能图标是否亮起。

```squirrel
function flushCommandEnable_职业名(obj)
{
    if (!obj) return S_FLOW_NORMAL;
    
    if (!obj.isInBattle())
    {
        sq_SetAllCommandEnable(obj, false);
        return S_FLOW_RETURN;
    }
    
    // 根据条件设置技能可用性
    obj.setSkillCommandEnable(SKILL_ID, true);
}
```

```squirrel
local skill = sq_GetSkill(obj, SKILL_ID);  // 获取技能对象
```

### 获取技能等级

```squirrel
local skillLevel = sq_GetSkillLevel(obj, SKILL_ID);  // 获取技能等级
```

```squirrel
// 方式1
local validTime = obj.sq_GetLevelData(SKILL_ID, index, skillLevel);
// 参数1：技能编号

// 参数3：技能等级

local validTime = sq_GetLevelData(obj, SKILL_ID, index, skillLevel);
```

```squirrel
// 方式1
local value = obj.sq_GetIntData(SKILL_ID, index);

// 参数2：静态数据第几个（从0开始）

// 方式2
local value = sq_GetIntData(obj, SKILL_ID, index);
```

## 技能伤害计算

### 获取百分比伤害

```squirrel
// 参数1：技能编号
// 参数2：状态编号
// 参数3：动态数据第几个
// 参数4：以多少比率获取（1.0表示100%）
local attackBonusRate = obj.sq_GetBonusRateWithPassive(
    SKILL_ID, STATE_ID, 0, 1.0
);
```

### 获取固伤

```squirrel
// 参数1：技能编号
// 参数2：状态编号
// 参数3：固伤动态数据第几个
// 参数4：百分比率动态数据第几个（如果只有固伤写-1）
// 参数5：以多少比率获取
local power = obj.sq_GetPowerWithPassive(
    SKILL_ID, STATE_ID, 1, 0, 1.0
);
```

### 设置伤害

```squirrel
// 设置百分比伤害
obj.sq_SetCurrentAttackBonusRate(attackBonusRate);

// 设置固伤
obj.sq_SetCurrentAttackPower(power);
```

## Buff 技能开发

### Buff技能基本结构

一个完整的Buff技能通常包含两部分：
1. 技能主体文件（`.nut`）- 处理技能触发和状态
2. Appendage文件（`ap_*.nut`）- 处理Buff持续效果

### checkExecutableSkill_* - Buff技能可执行性检查

```squirrel
function checkExecutableSkill_BuffZy(obj)
{
    if (!obj) return false;
    local isUse = obj.sq_IsUseSkill(SKILL_BUFFZY);  // 检查技能是否可使用

    if (isUse) {
        obj.sq_AddSetStatePacket(STATE_BUFFZY, STATE_PRIORITY_USER, false);
        return true;
    }

    return false;
}
```

### checkCommandEnable_* - Buff技能按键开关

```squirrel
function checkCommandEnable_BuffZy(obj)
{
    if (!obj) return false;
    local state = obj.sq_GetState();  // 获取当前角色的状态

    if (state == STATE_STAND)         // 站立状态可以使用
        return true;

    return true;
}
```

### onSetState_* - Buff技能状态设置

```squirrel
function onSetState_BuffZy(obj, state, datas, isResetTimer)
{
    if(!obj) return;

    obj.sq_StopMove();  // 停止移动
    obj.sq_SetCurrentAnimation(CUSTOM_ANI_STAGE_BLOODBOOM);
    obj.sq_SetStaticSpeedInfo(SPEED_TYPE_CAST_SPEED, SPEED_TYPE_CAST_SPEED,
        SPEED_VALUE_DEFAULT, SPEED_VALUE_DEFAULT, 1.0, 1.0);
```

### onEndCurrentAni_* - Buff技能动画结束后创建AP

```squirrel
function onEndCurrentAni_BuffZy(obj)
{
    if(!obj) return;

    local skillLevel = obj.sq_GetSkillLevel(SKILL_BUFFZY);
    local Time = obj.sq_GetLevelData(SKILL_BUFFZY, 0, skillLevel);

    // 创建AP
    local appendage = CNSquirrelAppendage.sq_AppendAppendage(
        obj, obj, SKILL_BUFFZY, false, 
        "character/swordman/buffzy/ap_buffzy.nut", false
    );

    // 设置左下角图标（必须）
    appendage.setAppendCauseSkill(
        BUFF_CAUSE_SKILL, 
        sq_getJob(obj),
        SKILL_BUFFZY,
        skillLevel
    );

    // 设定AP的持续时间（毫秒）
    appendage.sq_SetValidTime(Time);

    // 获取技能动态数据
    local attackspeed = obj.sq_GetLevelData(SKILL_BUFFZY, 1, skillLevel);
    local movespeed = obj.sq_GetLevelData(SKILL_BUFFZY, 2, skillLevel);
    local castspeed = obj.sq_GetLevelData(SKILL_BUFFZY, 3, skillLevel);

    // 存储数据到vector
    appendage.getVar("buffdata").clear_vector();
    appendage.getVar("buffdata").push_vector(attackspeed);
    appendage.getVar("buffdata").push_vector(movespeed);
    appendage.getVar("buffdata").push_vector(castspeed);

    // 附加AP（使用ID，不可重复）
    CNSquirrelAppendage.sq_AppendAppendageID(
        appendage, obj, obj, 
        APID_BUFF_ZY,  // APID常量
        true           // 是否为buff（true不能带过图）
    );

    obj.sq_AddSetStatePacket(STATE_STAND, STATE_PRIORITY_USER, false);
}
```

### Buff技能AP文件结构

```squirrel
// ap_buffzy.nut

APID_BUFF_ZY <- 230;  // 设置APID（可选）

function sq_AddFunctionName(appendage)
{
    appendage.sq_AddFunctionName("onStart", "onStart_appendage_BuffZy");
}

function onStart_appendage_BuffZy(appendage)
{
    if (!appendage) return;

    local obj = appendage.getParent();
    
    // 读取存储的数据（按代码先后顺序取值）
    local atkspeed = appendage.getVar("buffdata").get_vector(0);
    local movespeed = appendage.getVar("buffdata").get_vector(1);
    local castspeed = appendage.getVar("buffdata").get_vector(2);

    // 获取或创建属性状态
    local change_appendage = appendage.sq_getChangeStatus("BuffZy");
    
    // 如果不存在，创建属性状态
    if(!change_appendage)
    {
        change_appendage = appendage.sq_AddChangeStatusAppendageID(
            obj, obj, 0,
            CHANGE_STATUS_TYPE_ATTACK_SPEED,  // 属性类型
            false,                            // false为数值，true为百分比
            atkspeed,                         // 数值
            APID_COMMON                       // AppendageID
        );
    }
    
    // 设置属性参数
    if(change_appendage)
    {
        change_appendage.clearParameter();
        change_appendage.addParameter(CHANGE_STATUS_TYPE_ATTACK_SPEED, false, atkspeed.tofloat());
        change_appendage.addParameter(CHANGE_STATUS_TYPE_MOVE_SPEED, false, movespeed.tofloat());
        change_appendage.addParameter(CHANGE_STATUS_TYPE_CAST_SPEED, false, castspeed.tofloat());
    }
}
```

## 完整技能示例：血爆技能

### 技能文件结构

血爆技能包含三部分：
1. 加载文件（`swordman_load_state.nut`）- 注册技能和挂载特效
2. 技能主体文件（`BloodBoom.nut`）- 处理技能逻辑
3. 特效文件（`po_BloodBoom.nut`）- 处理特效逻辑

### 加载文件（swordman_load_state.nut）

```squirrel
// 挂载obj特效，编号要在24300到24400之间
IRDSQRCharacter.pushPassiveObj("Character/Swordman/BloodBoom/po_BloodBoom.nut", 24320);

// 注册技能状态
IRDSQRCharacter.pushState(
    ENUM_CHARACTERJOB_SWORDMAN, 
    "Character/Swordman/BloodBoom/BloodBoom.nut", 
    "BloodBoom", 
    STATE_BLOODBOOM, 
    SKILL_BLOODBOOM
);
```

### 技能主体文件（BloodBoom.nut）

```squirrel
// 技能可执行性检查
function checkExecutableSkill_BloodBoom(obj)
{
    if (!obj) return false;
    local isUse = obj.sq_IsUseSkill(SKILL_BLOODBOOM);  // 检查技能是否可使用

    if (isUse) {
        obj.getVar().setInt(FLAG_HIT, 0);  // 初始化标记
        obj.sq_AddSetStatePacket(STATE_BLOODBOOM, STATE_PRIORITY_USER, false);
        return true;
    }

    return false;
}

// 技能按键开关
function checkCommandEnable_BloodBoom(obj)
{
    if (!obj) return false;
    local state = obj.sq_GetState();

    if (state == STATE_STAND)  // 站立状态可以使用
        return true;

    return true;
}

// 设置状态
function onSetState_BloodBoom(obj, state, datas, isResetTimer)
{    
    if(!obj) return;

    obj.sq_StopMove();  // 停止移动
    obj.sq_SetCurrentAnimation(CUSTOM_ANI_STAGE_BLOODBOOM);  // 设置动画
    obj.sq_SetStaticSpeedInfo(SPEED_TYPE_CAST_SPEED, SPEED_TYPE_CAST_SPEED,
        SPEED_VALUE_DEFAULT, SPEED_VALUE_DEFAULT, 1.0, 1.0); 
}

// 状态持续处理
function onProc_BloodBoom(obj)
{
    if(!obj) return false;

    local currentAni = obj.getCurrentAnimation();  // 获取当前动画
    local currentAniIndex = currentAni.GetCurrentFrameIndex();  // 获取动画帧数
    local Hit = obj.getVar().getInt(FLAG_HIT);  // 获取标记，控制特效创建次数
    
    // 在第20帧且未创建特效时创建特效
    if(currentAniIndex == 20 && Hit == 0)
    {
        // 创建举手血珠特效（无实际伤害）
        obj.sq_SendCreatePassiveObjectPacket(38001, 0, 0, 0, 0);
        
        // 获取技能伤害
        local power = obj.sq_GetPowerWithPassive(SKILL_BLOODBOOM, STATE_BLOODBOOM, 1, -1, 1.0);
        local z = obj.getObjectHeight() / 2;
        
        // 开始写入数据
        obj.sq_StartWrite();
        obj.sq_WriteDword(power);
        
        // 创建PO特效（编号24320在24300-24400区间内，可以挂载nut并吃技能数据）
        obj.sq_SendCreatePassiveObjectPacket(24320, 0, 0, 0, z);
        
        // 设置标记+1，防止多次创建特效
        obj.getVar().setInt(FLAG_HIT, Hit + 1);
    }
    return true;
}

// 动画结束
function onEndCurrentAni_BloodBoom(obj)
{
    obj.getVar().setInt(FLAG_HIT, 0);  // 重置标记
    obj.sq_AddSetStatePacket(STATE_STAND, STATE_PRIORITY_USER, false);
}
```

### 特效文件（po_BloodBoom.nut）

```squirrel
// 设置PO特效状态
PASSIVEOBJ_SUB_STATE_0 <- 10  // 设置为1会卡死，应该从10开始
PASSIVEOBJ_SUB_STATE_1 <- 11

// 接收自定义数据（函数名po_BloodBoomFinish必须与特效列表中的名称一致）
function setCustomData_po_BloodBoomFinish(obj, reciveData)
{
    if(!obj) return;
    
    local damage = reciveData.readDword();  // 从缓存中读取伤害值

    // 清除并设置状态
    obj.getVar("state").clear_vector();
    obj.getVar("state").push_vector(PASSIVEOBJ_SUB_STATE_0);

    // 获取攻击信息并设置伤害
    local attackinfo = sq_GetCurrentAttackInfo(obj);
    sq_SetCurrentAttackPower(attackinfo, damage);  // 设置固定伤害

    // 如果是自己控制的对象，发送状态包
    if(obj.isMyControlObject()) {
        local pIntVec = sq_GetGlobalIntVector();  // 获取全局Int向量
        sq_IntVectorClear(pIntVec);  // 清空
        sq_IntVectorPush(pIntVec, 0);  // 插入值0
        obj.addSetStatePacket(PASSIVEOBJ_SUB_STATE_0, pIntVec, STATE_PRIORITY_AUTO, false, "");
    }
}

// 设置PO特效状态
function setState_po_BloodBoomFinish(obj, state, datas)
{
    if(!obj) return;

    local passiveState = state;
    obj.getVar("state").set_vector(0, passiveState);

    if(passiveState == PASSIVEOBJ_SUB_STATE_0)
    {
        // 获取特效的动画（0是[etc motion]下面的第一个）
        local ani = obj.getCustomAnimation(0);
        obj.setCurrentAnimation(ani);  // 设置当前动画
    }
}

// 动画结束
function onEndCurrentAni_po_BloodBoomFinish(obj)
{
    if(!obj) return;
    sq_SendDestroyPacketPassiveObject(obj);  // 销毁特效
}
```

### 关键技术点

1. **PO特效编号**：能挂载nut的特效编号要在24300到24400之间
2. **帧数控制**：使用 `obj.getVar().setInt(FLAG_HIT, value)` 控制特效创建次数，避免重复创建
3. **数据传输**：使用 `sq_StartWrite()` 和 `sq_WriteDword()` 写入数据，在PO中使用 `reciveData.readDword()` 读取
4. **伤害设置**：在PO中使用 `sq_SetCurrentAttackPower()` 设置固定伤害
5. **状态管理**：PO特效的状态不能设置为1，应从10开始避免冲突
